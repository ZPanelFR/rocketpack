Database schema (optional)

Ensure if you intend on using the database that the follow SQL has been run to create your database schema. The below schema provides some basic tables of which some of our core classes rely on to store system options and if you decide to enable, database sessions as a replacement for the standard PHP file sessions. 
    CREATE DATABASE IF NOT EXISTS `rocketpack` /*!40100 DEFAULT CHARACTER SET utf8 */;
    USE `rocketpack`;

    DROP TABLE IF EXISTS `t_session`;
    CREATE TABLE `t_session` (
        se_hash_vc` varchar(100) NOT NULL,
        `se_data_tx` text,
        `se_expires_ts` int(11) NOT NULL,
        PRIMARY KEY (`se_hash_vc`)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8;

    DROP TABLE IF EXISTS `t_setting`;
    CREATE TABLE `t_setting` (
        `so_name_vc` varchar(50) NOT NULL,
        `so_value_tx` text,
        `so_comment_tx` text,
        PRIMARY KEY (`so_name_vc`)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8;
